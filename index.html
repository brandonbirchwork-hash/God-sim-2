const canvas = document.getElementById("c"); const ctx = canvas.getContext("2d"); canvas.width = window.innerWidth; canvas.height = window.innerHeight; // ====================== // WORLD // ====================== const size = 50; const tile = 12; let food = []; // generate food for (let y = 0; y < size; y++) { let row = []; for (let x = 0; x < size; x++) { row.push(Math.random() < 0.2 ? 5 : 0); } food.push(row); } // ====================== // ENTITY // ====================== let x = 25; let y = 25; let energy = 30; // ====================== // MOVE TOWARD FOOD // ====================== function move() { let dirs = [[1,0],[-1,0],[0,1],[0,-1]]; let bestX = x; let bestY = y; let bestFood = food[y][x]; for (let d of dirs) { let nx = x + d[0]; let ny = y + d[1]; if (nx < 0 || ny < 0 || nx >= size || ny >= size) continue; if (food[ny][nx] > bestFood) { bestFood = food[ny][nx]; bestX = nx; bestY = ny; } } x = bestX; y = bestY; } // ====================== // LOOP // ====================== function loop() { ctx.fillStyle = "black"; ctx.fillRect(0,0,canvas.width,canvas.height); // draw world for (let yy = 0; yy < size; yy++) { for (let xx = 0; xx < size; xx++) { ctx.fillStyle = food[yy][xx] > 0 ? "#2ecc71" : "#0b2a4a"; ctx.fillRect(xx*tile, yy*tile, tile, tile); } } // energy drain energy -= 0.05; // eat if (food[y][x] > 0) { energy += 1; food[y][x] -= 0.3; } // death reset if (energy <= 0) { x = 25; y = 25; energy = 30; } // movement move(); // draw entity ctx.fillStyle = "white"; ctx.fillRect(x*tile, y*tile, tile, tile); // UI document.getElementById("ui").innerText = `God Sim - Stable Food System Energy: ${energy.toFixed(1)} Position: ${x}, ${y}`; requestAnimationFrame(loop); } loop();
